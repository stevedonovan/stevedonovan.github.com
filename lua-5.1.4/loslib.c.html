L0144        setfield(L, "wday", stm->tm_wday+1);
L0145        setfield(L, "yday", stm->tm_yday+1);
L0146        setboolfield(L, "isdst", stm->tm_isdst);
L0147      }

L0173        t = time(NULL);  /* get current time */

L0183        ts.tm_year = getfield(L, "year", -1) - 1900;
